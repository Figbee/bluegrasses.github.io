func root(w http.ResponseWriter, r *http.Request) {
    //1. 使用r.ParseForm()获取数据之后,get post 所有的数据都到了r.Form中
	r.ParseForm()
    //2.使用r.FormValue()从r.Form中检索数据
	fmt.Println(r.FormValue("name"))

}
func main() {
	http.HandleFunc("/", root)
	http.ListenAndServe(":9000", nil)
}
